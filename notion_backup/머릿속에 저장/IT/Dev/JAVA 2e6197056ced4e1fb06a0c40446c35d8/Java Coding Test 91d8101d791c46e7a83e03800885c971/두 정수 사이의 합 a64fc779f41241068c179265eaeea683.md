class Solution {
    public long solution(int a, int b) {
         long answer = 0;
        long result = 0;
        // -3 < 3
        if (a < b) {
            int c = b;
            b = a;
            a = c;
        }
        for (int idx = b; idx <= a; idx++) {
            if (a == b) {
                answer = a;

            } else {
                result = result + idx;
                answer = result;

            }
        }
        return answer;
    }
}
